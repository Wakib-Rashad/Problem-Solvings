                                              #include<stdlib.h>

                                              struct Maxheap;//forward declaration

                                              void resizeheap(struct Maxheap*);

                                              void perlocatedown(struct Maxheap*,int);

                                              typedef struct Maxheap

                                              {

                                              int *array;

                                              int count;

                                              int capacity;

                                              }maxheap;

                                              maxheap*createheap(int capacity)

                                              {

                                              maxheap*h=(maxheap*)malloc(sizeof(maxheap));

                                              if(h==NULL)

                                              {

                                              printf("Memory Error");

                                              return;

                                              }

                                              h->count=0;

                                              h->capacity=capacity;

                                              h->array=(int*)malloc(sizeof(int)*(h->capacity));

                                              if(h->array==NULL)

                                              {

                                              printf("Memory error");

                                              return;

                                              }

                                              return h;

                                              }

                                              //perlocate up

                                              void insert(maxheap*h,int data)

                                              {

                                              if(h->count==h->capacity)

                                              resizeheap(h);

                                              int i;

                                              h->count++;

                                              i=h->count-1;

                                              h->array[i]=data;

                                              int temp;

                                              while(i!=0 && h->array[i]>h->array[(i-1)/2])

                                              {

                                              temp=h->array[i];

                                              h->array[i]=h->array[(i-1)/2];

                                              h->array[(i-1)/2]=temp;;

                                              i=(i-1)/2;

                                              }

                                              }

                                              void swap(int *x,int *y)

                                              {

                                              int temp=*x;

                                              *x=*y;

                                              *y=temp;

                                              }

                                              void resizeheap(maxheap*h)

                                              {

                                              int i;

                                              int*array_old=h->array;

                                              h->array=(int*)malloc(sizeof(int)*(h->capacity*2));

                                              if(h->array==NULL)

                                              {

                                              printf("Invalid memmory");

                                              return;

                                              }

                                              for(i=0;i<h->capacity;i++)

                                              {

                                              h->array[i]=array_old[i];

                                              }

                                              h->capacity=2*h->capacity;

                                              free(array_old);

                                              }

                                              int parent(maxheap*h,int i)

                                              {

                                              if(i<=0|| i>=h->count)

                                              return -1;

                                              return (i-1)/2;

                                              }

                                              int leftchild(maxheap*h,int i)

                                              {

                                              int left=2*i+1;

                                              if(left<0 ||left>=h->count)

                                              return -1;

                                              return left;

                                              }

                                              int rightchild(maxheap*h,int i)

                                              {

                                              int right=2*i+2;

                                              if(right<0||right>=h->count)

                                              return -1;

                                              return right;

                                              }

                                              void print(maxheap*h)

                                              {

                                              int i;

                                              for(i=0;i<h->count;i++)

                                              printf("%d\n",h->array[i]);

                                              }

                                              int getmax(maxheap*h)

                                              {

                                              if(h->count==0)

                                              return -1;

                                              return h->array[0];

                                              }

                                              int deletemax(maxheap*h)

                                              {

                                              int data;

                                              if(h->count==0)

                                              return -1;

                                              data=h->array[0];

                                              h->array[0]=h->array[h->count-1];

                                              h->count--;

                                              perlocatedown(h,0);

                                              return data;

                                              }

                                              void perlocatedown(maxheap*h,int i)

                                              {

                                              int l,r,max,temp;

                                              l=leftchild(h,i);

                                              r=rightchild(h,i);

                                              if(l!=-1 && h->array[l]>h->array[i])

                                              max=l;

                                              else

                                              max=i;

                                              if(r!=-1 && h->array[r]>h->array[max])

                                              max=r;

                                              if(max!=i)

                                              {

                                              temp=h->array[i];

                                              h->array[i]=h->array[max];

                                              h->array[max]=temp;

                                              perlocatedown(h,max);

                                              }

                                              }

                                              void buildheap(maxheap*h,int a[],int n)

                                              {

                                              if(h==NULL)

                                              return;

                                              if(n>h->capacity)

                                              resizeheap(h);

                                              int i;

                                              for(i=0;i<n;i++) h-="">array[i]=a[i];

                                              h->count=n;

                                              for(i=(n-1)/2;i>=0;i--)

                                              perlocatedown(h,i);

                                              }

                                              void heapsort(int a[],int n)

                                              h->array[0]=h->array[h->count-1];

                                              h->count--;

                                              perlocatedown(h,0);

                                              }

                                              h->count=old_size;

                                                                    MinHeap Sort code

                                                                    #include<stdio.h>
                                                                    #include<stdlib.h>

                                                                    typedef struct Minheap
                                                                    {
                                                                    int count;
                                                                    int*array;
                                                                    int capacity;
                                                                    }minheap;

                                                                    minheap*createheap(int capacity)
                                                                    {
                                                                    minheap*h=(minheap*)malloc(sizeof(minheap));

                                                                    h->count=0;
                                                                    h->capacity=capacity;
                                                                    h->array=(int*)malloc((h->capacity)*sizeof(int));

                                                                    return h;
                                                                    }

                                                                    void insert(minheap*h,int data)
                                                                    {
                                                                    int i;
                                                                    if(h->count==h->capacity)
                                                                    resizeheap(h);
                                                                    h->count++;
                                                                    i=h->count-1;
                                                                    if(i==0)
                                                                    {
                                                                    h->array[i]=data;
                                                                    return;
                                                                    }
                                                                    while(i>=0 && data<h->array[(i-1)/2])
                                                                    {
                                                                    h->array[i]=h->array[(i-1)/2];
                                                                    i=(i-1)/2;
                                                                    if(i==0)
                                                                    break;
                                                                    }
                                                                    h->array[i]=data;
                                                                    }

                                                                    void resizeheap(minheap*h)
                                                                    {
                                                                    int i;
                                                                    int *arrayold=h->array;
                                                                    h->array=(int*)malloc(sizeof(int)*h->capacity*2);

                                                                    for(i=0;i<h->capacity;i++)
                                                                    {
                                                                    h->array[i]=arrayold[i];
                                                                    }
                                                                    h->capacity*=2;
                                                                    free(arrayold);
                                                                    }

                                                                    int leftchild(minheap*h,int i)
                                                                    {

                                                                    int left=2*i+1;
                                                                    if(left>=h->count)
                                                                    return -1;
                                                                    return left;
                                                                    }

                                                                    int rightchild(minheap*h,int i)
                                                                    {

                                                                    int right=2*i+2;
                                                                    if(right>=h->count)
                                                                    return -1;
                                                                    return right;
                                                                    }

                                                                    int parent(minheap *h,int i)
                                                                    {
                                                                    if(i<=0||i>=h->count)
                                                                    return -1;
                                                                    return (i-1)/2;
                                                                    }

                                                                    void deletemin(minheap*h)
                                                                    {
                                                                    int data;
                                                                    if(h->count==0)
                                                                    return;

                                                                    data=h->array[0];
                                                                    h->array[0]=h->array[h->count-1];
                                                                    h->count--;

                                                                    perlocatedown(h,0);
                                                                    }

                                                                    void perlocatedown(minheap*h,int i)
                                                                    {
                                                                    int l,r,min,temp;

                                                                    l=leftchild(h,i);
                                                                    r=rightchild(h,i);

                                                                    if(l!=-1 && h->array[l]<h->array[i])
                                                                    min=l;
                                                                    else
                                                                    min=i;

                                                                    if(r!=-1 && h->array[r]<h->array[min])
                                                                    min=r;

                                                                    if(min!=i)
                                                                    {
                                                                    temp=h->array[i];
                                                                    h->array[i]=h->array[min];
                                                                    h->array[min]=temp;
                                                                    perlocatedown(h,min);
                                                                    }
                                                                    return;
                                                                    }

                                                                    void print(minheap*h)
                                                                    {
                                                                    int i;
                                                                    for(i=0;i<h->count;i++)
                                                                    printf("\n%d",h->array[i]);

                                                                    printf("\n");
                                                                    }
