                                                                                                                                public class InsertionSort {

                                                                                                                                private static void insertionSort(int arr[],int len) {

                                                                                                                                for (int i = 0; i < len; i++) {

                                                                                                                                for (int j = i+1; j < (len-1); j++) {

                                                                                                                                if (arr[i] > arr[j]) {

                                                                                                                                arr[i] = arr[i] + arr[j];

                                                                                                                                arr[j] = arr[i] - arr[j];

                                                                                                                                arr[i] = arr[i] - arr[j];

                                                                                                                                }

                                                                                                                                }

                                                                                                                                }

                                                                                                                                }
