                      struct node
                      {
                      int data;
                      node* next;
                      };
                      void SeletSort(node* head)
                      {
                      node* p=head,*q,*k;
                      int tmp;
                      while(p!=NULL&&p->next!=NULL)
                      {
                      q=p->next;
                      k=p;
                      while(q!=NULL)
                      {
                      if(q->data < k->data)
                      k=q;
                      q=q->next;
                      }
                      tmp=p->data;
                      p->data=k->data;
                      k->data=tmp;
                      p=p->next;
                      }
                      }
                      }
